#### Volume:

#### Custom usage:

    docker run \
        -d \
        --name php-fpm \
        -v /path/www:/var/www \
        -e UPLOAD_MAX_FILESIZE=128M
        gists/php:7

#### Compose example:

```
version: '3'

services:
    php:
        container_name: php
        image: gists/php
        volumes:
            - /path/www:/var/www
            - phpsocket:/run/php
        environment:
            UPLOAD_MAX_FILESIZE=128M
        restart: always
    nginx:
        container_name: nginx
        image: nginx:stable-alpine
        ports:
            - "80:80"
            - "443:443"
        volumes:
            - /path/www:/var/www
            - phpsocket:/run/php
        depends_on:
            - php
        restart: always
volumes:
    phpsocket:
```

#### nginx configuration with php-fpm

with php7-fpm on gists/php:7

    fastcgi_pass    unix:/run/php/php7-fpm.sock;

with php5-fpm on gists/php:5:

    fastcgi_pass    unix:/run/php/php5-fpm.sock;
